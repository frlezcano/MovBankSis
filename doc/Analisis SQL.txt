			CAMPOS(est_che, fec_cobro, fec_entre)
			VALUES(  'I'  ,   NULL   ,    NULL  )
			WHERE (cod_banco  = cod_banco
            	   AND cod_ctacte = cod_ctacte
			   AND fec_ope    = fec_ope
			   AND no_che_tes = doc_ref
			   AND cod_ope    = cod_ope)

			CAMPOS(sal_depdia)
			VALUES(IFNULL(sal_depdia,0) + monto_gs
			WHERE (cod_banco = cod_banco
			   AND cod_ctacte= cod_ctacte)
